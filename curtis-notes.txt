    public function address_lookup(Request $request)
    {
        // Validate the incoming request parameters
        $request->validate([
            'address' => 'required|string',
            'zip_code' => 'required|string',
        ]);
    
        $street = $request->input('address');
        $zipCode = $request->input('zip_code');
        $apiKey = env('VITE_GOOGLE_MAP_API_KEY');
    
        if (!$apiKey) {
            return response()->json(['error' => 'Google Maps API key is not configured.'], 500);
        }
    
        $response = Http::post("https://addressvalidation.googleapis.com/v1:validateAddress?key=$apiKey", [
            'address' => [
                'regionCode' => 'US',
                'addressLines' =>$street,
            ],
            'enableUspsCass' => true,
        ]);
    
        if ($response->successful()) {
            $data = $response->json();
    
            if (!empty($data['results'])) {
                $results = [];
    
                foreach ($data['results'] as $result) {
                    $location = $result['address_components'];
                    $fullAddress = $result['formatted_address'] ?? 'Not found';
    
                    // Parse address components
                    $parsedData = [
                        'street' => 'Not found',
                        'city' => 'Not found',
                        'state_abbr' => 'Not found',
                        'zipcode' => 'Not found',
                        'code_suffix' => '',
                    ];
    
                    foreach ($location as $component) {
                        if (in_array('locality', $component['types'])) {
                            $parsedData['city'] = $component['long_name'];
                        }
                        if (in_array('administrative_area_level_1', $component['types'])) {
                            $parsedData['state_abbr'] = $component['short_name'];
                        }
                        if (in_array('postal_code', $component['types'])) {
                            $parsedData['zipcode'] = $component['long_name'];
                        }
                        if (in_array('postal_code_suffix', $component['types'])) {
                            $parsedData['code_suffix'] = $component['long_name'];
                        }
                        if (in_array('route', $component['types']) || in_array('street_number', $component['types'])) {
                            $parsedData['street'] = $component['long_name'];
                        }
                    }
    
                    // Combine ZIP code and suffix
                    if (!empty($parsedData['code_suffix'])) {
                        $parsedData['zipcode'] .= '-' . $parsedData['code_suffix'];
                    }
    
                    // Format the result
                    $results[] = [
                        'main_address' => $parsedData['street'],
                        'location' => $parsedData['city'] . ' ' . $parsedData['state_abbr'] . ' ' . $parsedData['zipcode'],
                    ];
                }
    
                return response()->json(['results' => $results]);
            }
    
            return response()->json(['data' => $data], 200);
            // return response()->json(['error' => 'No matching addresses found.'], 404);
        }
    
        return response()->json(['error' => 'Failed to fetch data from Google Maps API. Please try again later.'], 500);
    }
